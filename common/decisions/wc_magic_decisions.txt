drink_demon_blood_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_major_religion.dds"
	
	is_shown = {
		has_infusion_level_trigger = { MAGIC = arcane_fel AMOUNT = -1 }
		NOT = {
			has_trait = being_demon
			has_infusion_level_trigger = { MAGIC = arcane_fel AMOUNT = -3 }
		}
	}
	
	is_valid = {
		OR = {
			any_courtier = {
				has_trait = being_demon
			}
			any_close_family_member = {
				has_trait = being_demon
			}
		}
	}
	
	effect = {
		root = {
			trigger_event = WCINF.6001
			
			show_as_tooltip = {
				add_magic_infusion_effect = { MAGIC = arcane_fel ADD = -50 }
			}
		}
	}
	
	cooldown = { years = 1 }
	
	ai_check_interval = 0
}
