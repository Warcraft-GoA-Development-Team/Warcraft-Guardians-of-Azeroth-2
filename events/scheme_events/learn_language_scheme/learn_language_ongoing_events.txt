		scope:scheme ?= {
			scheme_progress > 1
		}
		is_travelling = yes 
		any_pool_character = {
			province = root.location 
			NOR = {
				this = scope:target
				has_relation_rival = root
			}
			is_available_ai_adult = yes
			culture = {
				has_same_culture_language = scope:target.culture
			}
		}
		exists = current_travel_plan.next_destination_province
	}
	
	immediate = {
		random_pool_character = {
			province = root.location
			limit = {
				NOR = {
					this = scope:target
					has_relation_rival = root
				}
				is_available_ai_adult = yes
				culture = {
					has_same_culture_language = scope:target.culture
				}
			}
			save_scope_as = interlocutor
		}
		current_travel_plan = {
			next_destination_province = { save_scope_as = next_destination }
		}
	}

	option = { # Have a lovely conversation
		name = learn_language_ongoing.1300.a
		flavor = learn_language_ongoing.1300.a.flavor
		scope:scheme = {
			add_scheme_progress = 1
		}
		stress_impact = {
			gregarious = miniscule_stress_impact_loss
		}
		ai_chance = {
			base = 50
		}
	}
	
	option = { # Try to engage in a debate
		name = learn_language_ongoing.1300.b
		flavor = learn_language_ongoing.1300.b.flavor
		duel = {
			skill = learning
			value = medium_skill_rating
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = 3.5
					min = -35
				}
				desc = learn_language_ongoing.1300.b.success
				send_interface_toast = {
					title = learn_language_ongoing.1300.b.success
					left_icon = root
					right_icon = scope:interlocutor
					scope:scheme = {
						add_scheme_progress = 2
					}
					scope:interlocutor.culture = {
						change_cultural_acceptance = {
							target = root.culture 
							value = minor_cultural_acceptance_gain
							desc = learn_lang_cult_acceptance_gain_debate
						}
					}
				}
			}
			50 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
					min = -35
				}
				desc = learn_language_ongoing.1300.b.failure
				send_interface_toast = {
					title = learn_language_ongoing.1300.b.failure
					left_icon = root
					right_icon = scope:interlocutor
					scope:scheme = {
						add_scheme_progress = -1
					}
				}
			}
		}
		
		stress_impact = {
			shy = miniscule_stress_impact_gain
			ambitious = minor_stress_impact_loss
			gregarious = miniscule_stress_impact_loss
		}
		ai_chance = {
			base = 20
		}
	}

	option = { # Join my entourage
		name = learn_language_ongoing.1300.c
		current_travel_plan = { add_companion = scope:interlocutor }
		if = {
			limit = {
				NOT = {
					scope:interlocutor = {
						is_courtier_of = root
					}
				}
			}
			add_courtier = scope:interlocutor
		}
		scope:scheme = {
			add_scheme_modifier = {
				type = interlocutor_entourage_member
			}
		}
		stress_impact = {
			gregarious = miniscule_stress_impact_loss
			trusting = miniscule_stress_impact_loss
			paranoid = miniscule_stress_impact_gain
		}
		ai_chance = {
			base = 30
			modifier = {
				has_trait = trusting
				add = 30
			}
		}
	}
}
