join_coalition_interaction = {
	interface_priority = 60
	category = interaction_category_diplomacy
	desc = invite_to_pol_faction_interaction_desc

	is_shown = {
		scope:actor = {
			can_be_in_coalition_trigger = yes
			NOT = { is_in_coaltion_against_trigger = { CRISIS_HOLDER = scope:recipient } }
		}
		scope:recipient = {
			is_crisis_trigger = yes
		}
	}
	
	is_valid_showing_failures_only = {
		scope:actor = {
			is_at_war = no
		}
	}
	
	on_accept = {
		scope:actor = {
			join_crisis_coalition_effect = { CRISIS_HOLDER = scope:recipient }
		}
	}
	auto_accept = yes

	ai_potential = {
		always = yes
	}
	ai_targets = {
		ai_recipients = neighboring_rulers
	}
	
	ai_frequency = 12
	ai_will_do = {
		base = 0
		modifier = {
			add = 100
			ai_has_coalition_reason_trigger = { CRISIS_HOLDER = scope:recipient }
		}
	}
}
