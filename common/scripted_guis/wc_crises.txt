get_scourge_crisis_title_gui = {
	scope = character
	is_shown = {
		always = yes
	}
	is_valid = {
		always = yes
	}
	effect = {
		clear_crisis_gui_effect = yes
		get_crisis_title_gui_effect = { TITLE = title:e_scourge }
	}
}

join_coalition_gui = {
	scope = character
	is_shown = {
		has_variable = crisis_character
		is_character_interaction_shown = {
			recipient = var:crisis_character
			interaction = join_coalition_interaction
		}
	}
	is_valid = {
		is_character_interaction_valid = {
			recipient = var:crisis_character
			interaction = join_coalition_interaction
		}
	}
	effect = {
		open_interaction_window = {
			interaction = join_coalition_interaction
			actor = root
			recipient = var:crisis_character
		}
	}
}
