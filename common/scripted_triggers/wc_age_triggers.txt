age_in_relation_to_humans_more_than_trigger = {
	save_temporary_scope_value_as = {
		name = temp_age
		value = {
			value = $age$
			if = {
				limit = { $age$ > 20 }
				multiply = racial_age_multiplier_value
			}
		}
	}
	
	age > scope:temp_age
}

race_relative_age_between_trigger = {
	age >= { value = scope:my_race_age_multiplier multiply = $AGE_1$ }
	age < { value = scope:my_race_age_multiplier multiply = $AGE_2$ }
}
