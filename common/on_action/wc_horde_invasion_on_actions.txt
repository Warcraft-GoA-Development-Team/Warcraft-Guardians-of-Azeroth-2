on_title_lost = {
	on_actions = {
		wc_invasion_on_actions
	}
}

wc_invasion_on_actions = {
	effect = {
		if = {
			limit = {
				scope:title = title:c_northshire
				scope:new_holder = { is_from_the_horde_trigger = yes }
			}
			scope:new_holder = {
				trigger_event = {
					id = wc_horde_invasion.0003
					days = 7
				}
			}
		}
	}
}
