scheme_hof_success_chance_modifier = {
	# Spiritual Head of Faith
	modifier = {
		add = $value$
		desc = "SCHEME_VS_SPIRITUAL_HOF"
		scope:target = {
			faith = scope:owner.faith
			faith = {
				has_doctrine = doctrine_spiritual_head
			}
			faith = {
				exists = religious_head
				religious_head = {
		 			this = scope:target
		 		}
			}
		}
	}
}

scheme_war_success_chance_modifier = {
	# At War
	first_valid = {
		modifier = {
			add = $value$
			desc = "SCHEME_AT_WAR"
			scope:target = {
				is_at_war_with = scope:owner
			}
		}
		modifier = {
			add = {
				value = $value$
				multiply = 0.5
			}
			scope:target = {
				is_landed = no
				exists = liege
				liege = {
					is_at_war_with = scope:owner
				}
			}
			desc = "SCHEME_AT_WAR_WITH_LIEGE"
		}
		modifier = {
			add = {
				value = $value$
				multiply = 0.75
			}
			scope:target = {
				is_landed = yes
				exists = liege
				liege = {
					is_at_war_with = scope:owner
				}
			}
			desc = "SCHEME_AT_WAR_WITH_LIEGE"
		}
	}
}
