			}

								AND = {
									has_government = feudal_government

										vassal_contract_obligation_level:feudal_government_taxes >= feudal_tax_normal_level

							OR = {

				value = 0
				if = {
					limit = {
						scope:vassal = {
							OR = {
								has_government = clan_government
								AND = {
									has_government = feudal_government
									OR = {
										vassal_contract_obligation_level:feudal_government_taxes < feudal_tax_normal_level

								}
