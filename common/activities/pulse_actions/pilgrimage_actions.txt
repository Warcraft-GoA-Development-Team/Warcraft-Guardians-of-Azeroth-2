#Pious Pilgrimage And Hajj

apa_prayer = { #Solitary Prayer
	#Either Pious Pilgrimage/Hajj or Reflection Intent if on Worldly Pilgrimage
	is_valid = {
		has_activity_type = activity_pilgrimage
		is_current_phase_active = yes
		OR = {
			has_activity_option = {
				category = special_type
				option = pilgrimage_type_pious
			}
			has_activity_option = {
				category = special_type
				option = pilgrimage_type_hajj
			}
			any_attending_character = {
				location = scope:province
				OR = {
					this = scope:host
					is_close_family_of = scope:host
					likes_character_trigger = { CHARACTER = scope:host }
				}
				trigger_if = {
					limit = {
						involved_activity = {
							has_activity_option = { category = special_type option = pilgrimage_type_worldly }
						}
					}
					has_activity_intent = reflection_intent
				}
			}
		}
	}

	weight = {
		value = 1
	}

	effect = {
		hidden_effect = {
			random_list = {
				10 = {
					trigger = {
						OR = {
							has_activity_option = {
								category = special_type
								option = pilgrimage_type_pious
							}
							has_activity_option = {
								category = special_type
								option = pilgrimage_type_hajj
							}
							activity_host = { has_activity_intent = reflection_intent }
						}
					}
					activity_host = { save_scope_as = first }
				}
				10 = {
					trigger = {
						any_attending_character = {
							location = scope:province
							OR = {
								this = scope:host
								is_close_family_of = scope:host
								likes_character_trigger = { CHARACTER = scope:host }
							}
							trigger_if = {
								limit = {
									involved_activity = {
										has_activity_option = { category = special_type option = pilgrimage_type_worldly }
									}
								}
								has_activity_intent = reflection_intent
							}
						}
					}
					random_attending_character = {
						limit = {
							location = scope:province
							OR = {
								this = scope:host
								is_close_family_of = scope:host
								likes_character_trigger = { CHARACTER = scope:host }
							}
							trigger_if = {
								limit = {
									involved_activity = {
										has_activity_option = {
											category = special_type
											option = pilgrimage_type_worldly
										}
									}
								}
								has_activity_intent = reflection_intent
							}
						}
						save_scope_as = first
					}
				}
			}
		}
		add_activity_log_entry = {
			key = apa_prayer
			tags = { pulse_action }
			character = scope:first
			
			scope:first = {
				add_piety = 15
				if = {
					limit = {
						has_trait = lifestyle_mystic
					}
					add_trait_xp = {
						trait = lifestyle_mystic
						value = lifestyle_mystic_xp_gain_minor_value
					}
				}
			}
		}
	}
}

apa_lead_procession = {

			}

		value = 1

			tags = { pulse_action }

			}

				}

					modifier = pilgrimage_apa_religious_procession_opinion
					target = scope:first
				}
			}
		}
	}

#Pious Pilgrimage or Hajj or Witch Ritual
apa_communal_prayer = {
	is_valid = {
		is_current_phase_active = yes
		any_attending_character = {
			location = scope:province
			is_adult = yes
			likes_character_trigger = { CHARACTER = scope:host }
			NOT = { this = scope:host }
		}
		OR = {
			AND = {
				has_activity_type = activity_pilgrimage
				OR = {
					has_activity_option = {
						category = special_type
						option = pilgrimage_type_pious
					}
					has_activity_option = {
						category = special_type
						option = pilgrimage_type_hajj
					}
				}
			}
			has_activity_type = activity_witch_ritual
		}
	}

	weight = {
		value = 1
	}

	effect = {
		scope:host = {
			save_scope_as = first
		}
		random_attending_character = {
			limit = {
				is_adult = yes
				location = scope:province
				likes_character_trigger = { CHARACTER = scope:host }
				NOT = { this = scope:first }
			}
			save_scope_as = second
		}

		add_activity_log_entry = {
			key = apa_communal_prayer
			tags = { pulse_action }
			character = scope:first
			target = scope:second
			
			scope:first = {
				add_opinion = {
					modifier = pilgrimage_apa_communal_prayer_opinion
					target = scope:second
				}
				add_piety = 15
				hidden_effect = {
					reverse_add_opinion = {
						modifier = pilgrimage_apa_communal_prayer_opinion
						target = scope:second
					}
				}
			}
			scope:second = {
				add_piety = 15
			}
		}
	}
}

#Reflection Intent
apa_silent_contemplation = {
	is_valid = {
		is_current_phase_active = yes
		OR = {
			AND = {
				has_activity_type = activity_pilgrimage
				any_attending_character = {
					location = scope:province
					has_activity_intent = reflection_intent
					OR = {
						this = scope:host
						is_close_family_of = scope:host
						likes_character_trigger = { CHARACTER = scope:host }
					}
				}
			}
			AND = {
				has_activity_type = activity_witch_ritual
				any_attending_character = {
					location = scope:province
					count >= 1
				}
			}
		}
	}

	weight = {
		value = 1
	}

	effect = {
		if = {
			limit = { has_activity_type = activity_pilgrimage }
			random_attending_character = {
				limit = {
					location = scope:province
					has_activity_intent = reflection_intent
					OR = {
						this = scope:host
						is_close_family_of = scope:host
						likes_character_trigger = { CHARACTER = scope:host }
					}
				}
				save_scope_as = first
			}
		}
		else = {
			random_attending_character = {
				limit = {
					location = scope:province
				}
				save_scope_as = first
			}
		}

		add_activity_log_entry = {
			key = apa_silent_contemplation
			tags = { pulse_action }
			character = scope:first
			
			scope:first = {
				add_piety = 10
				if = {
					limit = {
						has_trait = lifestyle_mystic
					}
					add_trait_xp = {
						trait = lifestyle_mystic
						value = lifestyle_mystic_xp_gain_minor_value
					}
				}
				stress_impact = {
					base = minor_stress_impact_loss
				}
			}
		}
	}
}

apa_preaching_to_unbelievers = {
	is_valid = {
		is_current_phase_active = yes
		scope:host = { #This intent is only available to the host
			location = scope:province
			has_activity_intent = zealot_intent 
			any_pool_character = {
				province = scope:province
				is_physically_able_ai_adult = yes
				NOT = { faith = scope:host.faith }
				NOT = { has_trait = zealous }
			}
		}
	}

	weight = {
		value = 0.1 #Intentionally lower weight
	}

	effect = {
		scope:host = {
			save_scope_as = first
		}
		random_pool_character = {
			province = scope:province
			limit = {
				NOT = { faith = scope:first.faith }
				NOT = { has_trait = zealous }
			}
			save_scope_as = second
		}

		add_activity_log_entry = {
			key = apa_preaching_to_unbelievers
			tags = { pulse_action }
			character = scope:first
			target = scope:second
			
			scope:second = {
				set_character_faith = scope:first.faith
			}
			scope:first = {
				add_piety = 15
			}
		}
		#Now we set a memory so that this conversion does not look to random if one looks at the character
		hidden_effect = { #Memory Management
			scope:first = {
				create_character_memory = {
					type = memory_apa_converted
					participants = {
						converted = scope:second
					}
				}
			}
			# Warcraft - Remove unused Hajj Variable throwing error
			scope:second = {
				create_character_memory = {
					type = memory_apa_was_converted
					participants = {
						converter = scope:first
					}
				}
			}
		}
	}
}

#Worldly Pilgrimages:
apa_cultured_discussion = {
	icon = public
	is_valid = {
		is_current_phase_active = yes
		has_activity_option = {
			category = special_type
			option = pilgrimage_type_worldly
		}
		activity_host = {
			location = scope:province
			NOT = {
				has_character_flag = cultural_acceptance_pilgrim
			}
		}
		exists = scope:province.province_owner.culture
		NOT = { activity_host = scope:province.province_owner }
		NOT = { activity_host.culture = scope:province.province_owner.culture }
	}

	weight = {
		value = 1
	}

	effect = {
		activity_host = {
			save_scope_as = first
		}
		scope:province.province_owner = {
			save_scope_as = second
		}

		add_activity_log_entry = {
			key = apa_cultured_discussion
			tags = { pulse_action }
			character = scope:first
			target = scope:second
			
			scope:first.culture = {
				change_cultural_acceptance = {
					target = scope:second.culture
					value = miniscule_cultural_acceptance_gain
					desc = pilgrimage_cultural_spread
				}
			}
		}
		hidden_effect = {
			scope:first = {
				add_character_flag = {
					flag = cultural_acceptance_pilgrim
					months = 2
				}
			}
		}
	}
}

#Random:
apa_pilgrim_hospitals = {
	icon = wound
	is_valid = {
		has_activity_type = activity_pilgrimage
		is_current_phase_active = yes
		any_attending_character = {
			location = scope:province
			OR = {
				has_trait = wounded_1
				has_trait = wounded_2
				has_trait = wounded_3
				has_trait = ill
			}
			OR = {
				this = scope:host
				is_close_family_of = scope:host
			}
		}
	}

	weight = { value = 5 } #Weighted up

	effect = {
		random_attending_character = {
			limit = {
				location = scope:province
				OR = {
					has_trait = wounded_1
					has_trait = wounded_2
					has_trait = wounded_3
					has_trait = ill
				}
				OR = {
					this = scope:host
					is_close_family_of = scope:host
				}
			}
			save_scope_as = first
		}

		add_activity_log_entry = {
			key = apa_pilgrim_hospitals
			tags = { pulse_action }
			character = scope:first
			
			scope:first = {
				if = {
					limit = {
						NOT = {
							has_trait = scarred
						}
						OR = {
							has_trait = wounded_1
							has_trait = wounded_2
							has_trait = wounded_3
						}
					}
					random = {
						chance = 40
						add_character_flag = will_get_scar
					}
				}
				switch = {
					trigger = has_trait
					wounded_1 = {
						remove_trait = wounded_1
					}
					wounded_2 = {
						change_trait_rank = {
							trait = wounded
							rank = -1
							max = 1
						}
					}
					wounded_3 = {
						change_trait_rank = {
							trait = wounded
							rank = -1
							max = 1
						}
					}
					ill = {
						remove_trait = ill
					}
				}
				if = {
					limit = {
						has_character_modifier = infected_wound_modifier
					}
					remove_character_modifier = infected_wound_modifier
				}
				if = {
					limit = {
						has_character_modifier = infected_wound_modifier
					}
					remove_character_modifier = infected_wound_modifier
					if = {
						limit = {
							NOR = {
								has_trait = scarred
								has_character_flag = will_get_scar
							}
						}
						add_character_flag = will_get_scar
					}
				}
				if = {
					limit = { has_character_flag = will_get_scar }
					add_trait = scarred
					add_trait_xp = {
		                trait = scarred
		                value = {
		                    integer_range = {
		                        min = 5
		                        max = 25
		                    }
		                }
		            }
					remove_character_flag = will_get_scar
				}
			}
		}
	}
}
