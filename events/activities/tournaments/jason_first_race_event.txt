scripted_trigger jason_first_race_event_1_opponent_trigger = {
	is_ai = yes
	exists = var:contest_aptitude
	NOT = { this = root }
	OR = {
		is_lowborn = no
		is_landed = no
	}
}

#Two racers collide with each other
jason_first_race_event.1 = {
	type = activity_event
	title = jason_first_race_event.1.t
	desc = {
		desc = jason_first_race_event.1.desc

			triggered_desc = {
				trigger = {
					OR = {
						has_trait = paranoid
						has_trait = vengeful

						has_relation_rival = scope:opponent

					right_icon = scope:opponent

			}

		stress_impact = {
			forgiving = medium_stress_impact_gain

			callous = miniscule_stress_impact_loss
			sadistic = miniscule_stress_impact_loss
		}

		ai_chance = {

			modifier = {
				factor = 0

					has_trait = forgiving
					has_trait = trusting
				}

			modifier = {
				add = -50

					has_trait = calm
				}

				add = -50

			modifier = {
				add = -25

				add = 100

				add = 100

					has_trait = vengeful

				add = 50

			modifier = {

				add = 100

			value = 10

					min = -49

						trait = tourney_participant

						value >= 20
					}

				}

					has_trait_xp = {
						trait = tourney_participant
						track = horse
						value >= 50

					add = 30
				}
				modifier = {
					has_trait_xp = {
						trait = tourney_participant
						track = horse
						value >= 80
					}
					add = 50
				}
				send_interface_toast = {
					title = jason_first_race_event.1.c.tt.win
					left_icon = root
					right_icon = scope:opponent

						type = successful_race_rescue

						limit = {

							trait = tourney_participant

			}
			#Your jockey can't hear you
			55 = {
				compare_modifier = {
					value = scope:duel_value
					multiplier = -3.5
					min = -49
				}

				send_interface_toast = {
					title = jason_first_race_event.1.c.tt.lose
					left_icon = root
					right_icon = scope:opponent
					scope:activity = {
						add_activity_log_entry = {
							key = contest_horse_race_collision_log
							tags = { contest bad }
							score = 30

							target = scope:opponent

				desc = jason_first_race_event.1.c.tt.lose

					type = trampled_during_race_rescue

			}

			craven = major_stress_impact_gain
			sadistic = medium_stress_impact_gain

			base = 100

					has_trait = sadistic

				add = -50

					has_trait = vengeful
				}
			}
			modifier = {
				add = -50
				OR = {
					diplomacy < mediocre_skill_rating
					prowess < mediocre_skill_rating
				}
			}
			modifier = {
				add = -25

				add = -50

			modifier = {
				add = -50
				highest_held_title_tier = tier_empire
			}
			modifier = {
				add = 75
				OR = {
					has_trait = compassionate
					has_trait = brave
				}
			}
			modifier = {
				add = 25
				has_trait_xp = {
					trait = tourney_participant
					track = horse
					value >= 50
				}
			}
			modifier = {
				add = 50
				has_trait_xp = {
					trait = tourney_participant
					track = horse
					value >= 100
				}

			modifier = {
				add = 50

					diplomacy >= very_high_skill_rating
					prowess >= very_high_skill_rating

		flavor = jason_first_race_event.1.d.tt
		scope:activity = {

				tags = { contest bad }

		}

		stress_impact = {
