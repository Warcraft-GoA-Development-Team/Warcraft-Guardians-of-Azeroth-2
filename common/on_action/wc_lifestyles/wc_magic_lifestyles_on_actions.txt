magic_lifestyles_progress_pulse = { # Fires 4 times per year
	trigger = {
		OR = {
			has_lifestyle = light_magic_lifestyle
			has_lifestyle = shadow_magic_lifestyle
			has_lifestyle = disorder_magic_lifestyle
			has_lifestyle = order_magic_lifestyle
			has_lifestyle = life_magic_lifestyle
			has_lifestyle = death_magic_lifestyle
			has_lifestyle = elemental_fire_magic_lifestyle
			has_lifestyle = elemental_water_magic_lifestyle
			has_lifestyle = elemental_air_magic_lifestyle
			has_lifestyle = elemental_earth_magic_lifestyle
		}
	}

	effect = {
		if = {
			limit = { has_focus_light_magic_trigger = yes }
			wc_add_MAGIC_experience_effect = { MAGIC = light EXP = wc_magic_lifestyle_pulse_exp_regular_value }
		}
		else_if = {
			limit = { has_focus_shadow_magic_trigger = yes }
			wc_add_MAGIC_experience_effect = { MAGIC = shadow EXP = wc_magic_lifestyle_pulse_exp_regular_value }
		}
		else_if = {
			limit = { has_focus_disorder_magic_trigger = yes }
			wc_add_MAGIC_experience_effect = { MAGIC = disorder EXP = wc_magic_lifestyle_pulse_exp_regular_value }
		}
		else_if = {
			limit = { has_focus_order_magic_trigger = yes }
			wc_add_MAGIC_experience_effect = { MAGIC = order EXP = wc_magic_lifestyle_pulse_exp_regular_value }
		}
		else_if = {
			limit = { has_focus_life_magic_trigger = yes }
			wc_add_MAGIC_experience_effect = { MAGIC = life EXP = wc_magic_lifestyle_pulse_exp_regular_value }
		}
		else_if = {
			limit = { has_focus_death_magic_trigger = yes }
			wc_add_MAGIC_experience_effect = { MAGIC = death EXP = wc_magic_lifestyle_pulse_exp_regular_value }
		}
		else_if = {
			limit = { has_focus_elemental_magic_trigger = yes }

			if = {
				limit = { has_focus_elemental_neither_spirit_nor_decay_magic_trigger = yes }

				if = {
					limit = { has_focus_elemental_fire_magic_trigger = yes }
					wc_add_MAGIC_experience_effect = { MAGIC = elemental_fire EXP = wc_magic_lifestyle_pulse_exp_elemental_regular_value }
				}
				else_if = {
					limit = { has_focus_elemental_water_magic_trigger = yes }
					wc_add_MAGIC_experience_effect = { MAGIC = elemental_water EXP = wc_magic_lifestyle_pulse_exp_elemental_regular_value }
				}
				else_if = {
					limit = { has_focus_elemental_air_magic_trigger = yes }
					wc_add_MAGIC_experience_effect = { MAGIC = elemental_air EXP = wc_magic_lifestyle_pulse_exp_elemental_regular_value }
				}
				else_if = {
					limit = { has_focus_elemental_earth_magic_trigger = yes }
					wc_add_MAGIC_experience_effect = { MAGIC = elemental_earth EXP = wc_magic_lifestyle_pulse_exp_elemental_regular_value }
				}
			}
			else = {
				if = {
					limit = { has_focus_elemental_spirit_magic_trigger = yes }
					wc_add_MAGIC_experience_effect = { MAGIC = elemental_spirit EXP = wc_magic_lifestyle_pulse_exp_shared_elemental_spirit_and_decay_value }
				}
				else_if = {
					limit = { has_focus_elemental_decay_magic_trigger = yes }
					wc_add_MAGIC_experience_effect = { MAGIC = elemental_decay EXP = wc_magic_lifestyle_pulse_exp_shared_elemental_spirit_and_decay_value }
				}

				if = {
					limit = { has_focus_elemental_fire_magic_trigger = yes }
					wc_add_MAGIC_experience_effect = { MAGIC = elemental_fire EXP = wc_magic_lifestyle_pulse_exp_shared_elemental_other_value }
				}
				else_if = {
					limit = { has_focus_elemental_water_magic_trigger = yes }
					wc_add_MAGIC_experience_effect = { MAGIC = elemental_water EXP = wc_magic_lifestyle_pulse_exp_shared_elemental_other_value }
				}
				else_if = {
					limit = { has_focus_elemental_air_magic_trigger = yes }
					wc_add_MAGIC_experience_effect = { MAGIC = elemental_air EXP = wc_magic_lifestyle_pulse_exp_shared_elemental_other_value }
				}
				else_if = {
					limit = { has_focus_elemental_earth_magic_trigger = yes }
					wc_add_MAGIC_experience_effect = { MAGIC = elemental_earth EXP = wc_magic_lifestyle_pulse_exp_shared_elemental_other_value }
				}
			}
		}
	}
}
