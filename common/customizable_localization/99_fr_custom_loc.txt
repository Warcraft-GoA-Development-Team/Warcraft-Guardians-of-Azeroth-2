FR_gender_fake = { # all gender custom commands are based on it
	type = character
	log_loc_errors = no
	text = {
		trigger = {
			trigger_if = { # one character (Custom)
				limit = { 
					exists = this
					NOT = { exists = scope:second }
				}
				this = { is_female = yes }
			}
			trigger_if = { # two characters (Custom2)
				limit = { 
					exists = this
					exists = scope:second
				}
				this = { is_female = yes }
				scope:second = { is_female = yes }
			}
		}
		localization_key = CustomLoc_FR_female_
	}
	text = {
		localization_key = CustomLoc_FR_male_
		fallback = yes
	}
}
