MAGIC_is_shunned_or_illegal_trigger = {
	faith = {
		OR = {
			has_doctrine_parameter = $MAGIC$_magic_illegal
			has_doctrine_parameter = $MAGIC$_magic_shunned
		}
	}
}

has_collected_souls_amount_trigger = {
	custom_description = {
		text = "HAS_SOULS"
		value = $AMOUNT$
		
		trigger_if = {
			limit = {
				has_variable = collected_souls
			}
			
			AND = {
				$AMOUNT$ > 0
				var:collected_souls >= $AMOUNT$
			}
		}
		trigger_else = {
			always = no
		}
	}
}
