mp_pause_travel_plan = {
	hidden_effect = {
		current_travel_plan ?= {
			pause_travel_plan_mp = yes
		}
	}
}

mp_resume_travel_plan = {
	hidden_effect = {
		if = {
			limit = { has_multiple_players = yes }
			current_travel_plan ?= {
				resume_travel_plan = yes
			}
		}
	}
}

mp_delay_travel_plan = {
	hidden_effect = {
		if = {
			limit = { has_multiple_players = yes }
			current_travel_plan ?= {
				delay_travel_plan = {
					days = $DAYS$
				}
			}
		}
	}
}
