meritocracy_perk = {
	lifestyle = stewardship_lifestyle
	tree = duty
	position = { 1 0 }
	icon = node_stewardship
	
	auto_selection_weight = {
		value = 11
		if = {
			limit = {
				has_education_stewardship_trigger = yes
			}
			add = 1989
		}
		if = {
			limit = {
				has_focus = stewardship_duty_focus
			}
			multiply = 5
		}
		if = {
			limit = {
				
				# Warcraft
				NOT = { can_spend_perks_in_nonelemental_TYPE_lifestyle_tree_TREE_trigger = { TYPE = stewardship TREE = 3 } }

			}
			multiply = 0
		}
	}
	
	effect = {
		# Warcraft
		trigger_event = { on_action = on_vanilla_lifestyle_perk_gained }
		if = {
			limit = {
				NOT = { government_has_flag = government_is_landless_adventurer }
			}
			custom_description_no_bullet = {
				text = meritocracy_perk_effect
			}
		}
	}

	character_modifier = {
		courtier_and_guest_opinion = 15
	}
}
