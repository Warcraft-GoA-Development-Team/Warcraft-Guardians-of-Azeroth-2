female_fertility_age_value = {
	value = 45
	if = {
		limit = {
			has_trait = fecund
		}
		add = 5
	}
	if = {
		limit = {
			exists = dynasty
			dynasty = {
				has_dynasty_perk = blood_legacy_5
			}
		}
		add = 5
	}
	multiply = racial_age_multiplier_value
}
