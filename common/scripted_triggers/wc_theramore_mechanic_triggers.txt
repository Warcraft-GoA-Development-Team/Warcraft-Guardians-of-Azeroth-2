is_theramorean_descendant_culture = {
	culture = {
		OR = {
			has_cultural_pillar = heritage_arathi
			has_cultural_pillar = heritage_gilnean
		}
		is_culture_or_parent_culture_trigger = { CULTURE = culture:theramore }
	}
}

has_this_law_in_te = {
	top_liege.primary_title = {
		has_variable_list = list_of_flag_laws
		is_target_in_variable_list = {
			name = list_of_flag_laws
			target = flag:law_$policy$_$number$
		}
	}
}
is_second_law_in_te = {
	top_liege.primary_title = {
		OR = {
			NOT = { has_variable_list = list_of_flag_laws }
			AND = {
				has_variable_list = list_of_flag_laws
				NOT = {
					is_target_in_variable_list = {
						name = list_of_flag_laws
						target = flag:law_$policy$_1
					}
				}
				NOT = {
					is_target_in_variable_list = {
						name = list_of_flag_laws
						target = flag:law_$policy$_3
					}
				}
			}
		}
	}
}
has_any_policy_with_number = {
	has_variable_list = list_of_flag_laws
	OR = {
		is_target_in_variable_list = {
			name = list_of_flag_laws
			target = flag:law_trade_$number$
		}
		is_target_in_variable_list = {
			name = list_of_flag_laws
			target = flag:law_union_$number$
		}
		is_target_in_variable_list = {
			name = list_of_flag_laws
			target = flag:law_culture_$number$
		}
		is_target_in_variable_list = {
			name = list_of_flag_laws
			target = flag:law_faith_$number$
		}
	}
}

tier_or_family_member_tier_equal_trigger = {
	OR = {
		highest_held_title_tier = $TIER$
		any_spouse = {
			highest_held_title_tier = $TIER$
		}
		any_close_family_member = {
			highest_held_title_tier = $TIER$
		}
	}
}
tier_or_family_member_tier_greater_or_equal_trigger = {
	OR = {
		highest_held_title_tier >= $TIER$
		any_spouse = {
			highest_held_title_tier >= $TIER$
		}
		any_close_family_member = {
			highest_held_title_tier >= $TIER$
		}
	}
}
