time_of_year_basic = {
	assert = { current_date = "1066.09.15" }
	expect = {
		time_of_year = {
			min = 08.01
			max = 11.15
		}
	}
}

time_of_year_exact_date = {
	assert = { current_date = "1066.09.15" }
	expect = {

			max = 09.15
		}
	}
}

		}
	}
