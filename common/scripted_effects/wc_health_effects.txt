check_pregnancy_effect = {
	if = {
		limit = {
			can_have_children = no
			is_pregnant = yes
		}
		end_pregnancy = yes
		pregnancy_cleanup_effect = yes
	}
}

hurt_or_wound_effect = { 
	random_list = { 
		60 = { increase_wounds_effect = { REASON = burned } }
		10 = {
			trigger = {
				NOT = { has_trait = maimed }
			}
			add_trait = maimed
		}
		10 = {
			trigger = {
				NOT = { has_trait = one_eyed }
			}
			add_trait = one_eyed
		}
		10 = {
			trigger = {
				NOT = { has_trait = one_legged }
			}
			add_trait = one_legged
		}
		5 = { 
			trigger = { 
				NOT = { has_trait = disfigured }
			}
			add_trait = disfigured
		}
		5 = { 
			if = { limit  = { is_female = yes}
				add_trait = infertile
			}
			else = { add_trait = eunuch }
		}
	}
}
