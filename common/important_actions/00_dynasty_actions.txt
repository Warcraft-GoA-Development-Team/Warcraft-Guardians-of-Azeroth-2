action_can_choose_dynasty_perk = {
	type = alert
	check_create_action = {
		if = {
			limit = {
				dynasty.dynast = this
				dynasty = {
					dynasty_can_unlock_relevant_perk = yes
				}
			}

			try_create_important_action = {
				important_action_type = action_can_choose_dynasty_perk
			}
		}
	}

	effect = {
		dynasty = {
			open_view_data = {
				view = dynasty_view
			}
		}
	}
}
