# Learning the lithurgical language

scripted_effect bp2_learn_lithurgical_language_effect = {
	if = {
		limit = {
			scope:activity.activity_location.county.faith.religion = religion:light_group
			NOT = { scope:activity.activity_location.county.faith = faith:khazism }
		}
		learn_language = language_common
	}
	else_if = {
		limit = {
			scope:activity.activity_location.county.faith.religion = faith:kaldorei_religion.religion
		}
		learn_language = language_darnassian
	}
	else_if = {
		limit = {
			scope:activity.activity_location.county.faith.religion = religion:arcane_group
		}
		learn_language = language_thalassian
	}
	else_if = {
		limit = {
			scope:activity.activity_location.county.faith.religion = religion:deathbringer
		}
		learn_language = language_cursed
	}
	else_if = {
		limit = {
			scope:activity.activity_location.county.faith.religion = religion:shadow_group
		}
		learn_language = language_gutterspeak
	}
	else_if = {
		limit = {
			OR = {
				scope:activity.activity_location.county.faith = faith:fire_cult
				scope:activity.activity_location.county.faith = faith:khazism
			}
		}
		learn_language = language_dwarven
	}
