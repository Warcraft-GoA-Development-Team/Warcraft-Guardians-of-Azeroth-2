	lifestyle = intrigue_lifestyle
	tree = skulduggery
	position = { 2 0 }
	icon = node_intrigue

	auto_selection_weight = {
		value = 11
		if = {
			limit = {
				has_education_intrigue_trigger = yes
			}
			add = 1989
		}
		if = {
			limit = {
				has_focus = intrigue_skulduggery_focus
			}

		}

	can_be_picked = {
		always = yes

	effect = {

		}
	}
}
