unleash_rage = {
	desc = unleash_rage_desc
	ai_check_interval = 1
	cooldown = { days = 182 }

	is_shown = {
		has_perk = unleashed_rage

		NOT = {
			has_character_modifier = unleashed_rage_modifier
		}
	}

	effect = {
		add_character_modifier = {
			modifier = unleashed_rage_modifier
			days = 30
		}
	}
}

reign_in_rage = {
	desc = reign_in_rage_desc
	ai_check_interval = 1
	is_shown = {
		has_perk = unleashed_rage

		has_character_modifier = unleashed_rage_modifier
	}

	effect = {
		remove_character_modifier = unleashed_rage_modifier
	}
}

go_running = {
	desc = go_running_desc
	ai_check_interval = 1
	cooldown = { years = 1 }

	is_shown = {
		has_perk = athletical
	}

	is_valid = {
		NOT = {
			has_trait = lazy
		}
	}

	effect = {
		add_character_modifier = {
			modifier = running_modifier
			years = 5
		}

		if = {
			limit = { current_weight >= 25 }
			change_current_weight = -10
		}

		if = {
			limit = { has_trait = athletic }
			add_stress = medium_stress_impact_loss
		}
	}
}
