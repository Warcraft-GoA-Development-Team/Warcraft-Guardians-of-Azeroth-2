MAGIC_lifestyle_is_valid_trigger = {
	trigger_if = {
		limit = { can_openly_learn_$MAGIC$_magic_trigger = no }

		OR = {
			$MAGIC$_magic_lifestyle_perks > 0
			$MAGIC$_magic_lifestyle_perk_points > 0
			$MAGIC$_magic_lifestyle_xp > 0	
		}
	}
}
