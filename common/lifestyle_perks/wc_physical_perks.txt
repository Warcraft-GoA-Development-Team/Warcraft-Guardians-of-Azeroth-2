pain_resistance = {
	lifestyle = physical_lifestyle
	tree = physical_tree_1
	position = { 2 0 }
	icon = node_martial

	auto_selection_weight = {
		#Basically for this one I'm trying to make sure people who fit the requirements for the other two and are in somewhat 
		#bad shape pick this one
		value = wc_physical_lifestyle_auto_selection_weight_base_value

		if = { #Base
			limit = { health_tree_auto_selection_weight_add_base_trigger = yes }
			add = wc_physical_lifestyle_auto_selection_weight_add_to_base_value
		}

		if = {
			limit = {
				OR = {
					has_perk = nightstalker
					has_perk = fortifying_feast
				}
			}
			add = wc_physical_lifestyle_auto_selection_weight_other_trees_add_to_base_value
		}
		else = {
			multiply = 0.75
		}

		if = {
			limit = { has_focus = physical_focus_1 }
			multiply = wc_magic_lifestyle_perk_auto_selection_weight_multiply_from_focus_value
		}
	}

	character_modifier = {
		stress_gain_mult = -0.1
		negate_health_penalty_add = 0.25
		negate_fertility_penalty_add = 0.1
	}
}

nightstalker = {
	lifestyle = physical_lifestyle
	tree = physical_tree_2
	position = { 1 0 }
	icon = node_martial

	auto_selection_weight = {
		value = wc_physical_lifestyle_auto_selection_weight_base_value

		if = { #Base
			limit = { dexterity_tree_auto_selection_weight_add_base_trigger = yes }
			add = wc_physical_lifestyle_auto_selection_weight_add_to_base_value
		}

		if = { #Education
			limit = { has_trait = education_intrigue }
			add = wc_physical_lifestyle_auto_selection_weight_education_intrigue_add_to_base_value
		} 
		else_if = {
			limit = {
				AND = {
					has_trait = education_martial_prowess
					is_hunter_trigger = yes
				}
			}
			add = wc_physical_lifestyle_auto_selection_weight_education_martial_and_hunter_add_to_base_value
		}

		if = {
			limit = { has_focus = physical_focus_2 }
			multiply = wc_magic_lifestyle_perk_auto_selection_weight_multiply_from_focus_value
		}
	}

	effect = {
		custom_description_no_bullet = {
			text = nightstalker_perk_effect
		}
	}
}

fortifying_feast = {
	lifestyle = physical_lifestyle
	tree = physical_tree_3
	position = { 1 0 }
	icon = node_martial

	auto_selection_weight = {
		value = wc_physical_lifestyle_auto_selection_weight_base_value

		if = { #Base
			limit = { might_tree_auto_selection_weight_add_base_trigger = yes }
			add = wc_physical_lifestyle_auto_selection_weight_add_to_base_value
		}

		if = { #Education
			limit = { has_trait = education_martial_prowess }
			add = wc_physical_lifestyle_auto_selection_weight_education_martial_add_to_base_value
		}
		else_if = {
			limit = {
				AND = {
					has_trait = education_martial
					prowess > 10
				}
			}
			add = wc_physical_lifestyle_auto_selection_weight_education_martial_add_to_base_value
		}
		else_if = {
			limit = {
				AND = {
					has_trait = education_martial
				}
			}
			add = wc_physical_lifestyle_auto_selection_weight_education_martial_add_to_base_lesser_value
		}

		if = {
			limit = { has_focus = physical_focus_3 }
			multiply = wc_magic_lifestyle_perk_auto_selection_weight_multiply_from_focus_value
		}
	}

	effect = {
		custom_description_no_bullet = {
			text = fortifying_feast_perk_effect
		}
	}
}
