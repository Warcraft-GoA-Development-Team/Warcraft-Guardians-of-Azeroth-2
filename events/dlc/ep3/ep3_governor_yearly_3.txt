scripted_trigger ep3_governor_yearly_3002_tax_farmer_trigger = {
	age >= age_18_value
	is_available = yes
	save_temporary_scope_as = tax_farmer
	root.faith = {
		has_dominant_ruling_gender = scope:tax_farmer
	}
	NOT = {
		this = root.cp:councillor_court_chaplain
	}
}
