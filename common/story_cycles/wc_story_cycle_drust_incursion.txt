story_drust_incursion = {	
	on_setup = {
		if = {
			limit = {
				NOT = {
					is_target_in_global_variable_list = {
						name = unavailable_unique_decisions
						target = flag:summon_gorak_tul 
					}
				}
			}
			add_to_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:summon_gorak_tul
			}
		}
		story_owner = {
			save_scope_as = gorak_tul
			var:summoner ?= {
				save_scope_as = summoner
			}
		}
		
		# Spawn troops
		story_owner = {
			trigger_event = {
				id = drust_crisis.5
				days = 2
			}
		}
	}
	
	on_owner_death = {
		end_story = yes
	}
}
