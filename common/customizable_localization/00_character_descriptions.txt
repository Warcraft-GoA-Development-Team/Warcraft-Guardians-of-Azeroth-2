RelationDescriptor = {
	type = character

	text = {
		trigger = {
			opinion = {
				target = scope:second
				value > 0
			}
		}
		localization_key = positive_desc_of_other
	}

	text = {
		trigger = {
			opinion = {
				target = scope:second
				value <= 0
			}
		}
		localization_key = negative_desc_of_other
	}
}
