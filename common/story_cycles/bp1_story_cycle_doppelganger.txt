story_doppelganger = {
	# by James Beaumont
	
	on_setup = {
		
	}
#
	on_end = {
		debug_log = "Doppelganger story ended on:"
		debug_log_date = yes
	}
#
	on_owner_death = {
		scope:bp1_yearly_9007_doppelganger = {
			silent_disappearance_effect = yes
		}
	}

	# Doppelganger dies, end the story
	effect_group = {
		days = { 40 60 }

		trigger = {
			exists = var:doppelganger_character
		}

		first_valid = {
			triggered_effect = {
				trigger = {
					exists = var:doppelganger_character
					var:doppelganger_character = {
						is_alive = no
					}
				}
				effect = {
					end_story = yes
				}
			}
		}
	}

	# Something happens - random events
	effect_group = {
		days = { 365 600 }
		chance = 50

		trigger = {
			exists = var:doppelganger_character
		}

		first_valid = {
			triggered_effect = {
				trigger = { always = yes }
				effect = {
					story_owner = {
						trigger_event = {
							on_action = ongoing_doppelganger_events
						}
					}
				}
			}
		}
	}
}
