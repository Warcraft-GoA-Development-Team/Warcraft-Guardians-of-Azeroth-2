	flag = please_dont_show_me_in_gui
	
	theramore_laws_culture_1 = {
		modifier = {
			county_opinion_add = 15
		}

		can_keep = {
			trigger_if = {
				limit = {
					top_liege ?= { is_theramorean_descendant_culture = yes }
				}
				top_liege ?= {
					primary_title = {
						has_variable_list = list_of_flag_laws
						is_target_in_variable_list = {
							name = list_of_flag_laws
							target = flag:law_culture_1
						}
					}
				}
			}
			is_theramorean_descendant_culture = yes
		}
		can_pass = {
			always = no
		}
		should_show_for_title = { always = no }
	}
