scripted_trigger jason_locale_events_100_mad_trigger = {
	OR = {
		has_trait = lunatic
		has_trait = berserker
	}
}

scripted_trigger jason_locale_events_100_demanding_trigger = {
	OR = {
		has_trait = greedy
		has_trait = ambitious
		has_trait = arrogant
		has_trait = disloyal
		has_trait = stubborn
	}
	NOR = {
		has_trait = diligent
		has_trait = loyal
		has_trait = humble
		has_trait = content
	}
}

scripted_trigger jason_locale_events_100_intoxicated_trigger = {
	OR = {
		has_trait = drunkard
		has_trait = hashishiyah
	}
	NOT = {

	}
