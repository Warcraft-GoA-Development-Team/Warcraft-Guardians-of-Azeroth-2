namespace = gorian_restoration

gorian_restoration.0001 = {
	type = character_event
	title = gorian_restoration.0001.t
	desc = {
		desc = gorian_restoration.0001.desc.intro
		# Doing it as the Ogres.
		triggered_desc = {
			trigger = {
				root.culture = culture:ogre
			}
			desc = gorian_restoration.0001.desc.ogre
		}
		# Doing it as moknathal
		triggered_desc = {
			trigger = {
				root.culture = culture:moknathal
			}
			desc = gorian_restoration.0001.desc.moknothal
		}
	}

	theme = realm
	left_portrait = {
		character = scope:scoped_emperor
		animation = personality_honorable
	}

	immediate = {
		play_music_cue = "mx_cue_epic_sacral_moment"
		save_scope_as = scoped_emperor
	}
	option = {
		name = gorian_restoration.0001.a
		set_nickname_effect = { NICKNAME = nick_the_glorious }
		# Send narrative fluff to other players (if any).
		every_player = {
			limit = {
				NOT = { this = root }
				is_within_diplo_range = { CHARACTER = root }
			}
			trigger_event = gorian_restoration.0002
		}
	}
}
