on_birthday = {
	on_actions = {
		on_birthday_aging
		on_birthday_weight
	}
}
on_birthday_weight = {
	trigger = {
		current_weight >= 10
		OR = {
			has_trait_with_flag = elven_family
			has_trait_with_flag = dragon_family
			has_trait = creature_sea_giant
		}
	}

	effect = {
		if = {
			limit = {
				target_weight >= 40
			}
			change_target_weight = -40
		}
		change_current_weight = -5
	}
}
