random_nickname_chance = {
	value = primary_title.tier
	add = {
		value = age
		#Warcraft
		divide = racial_age_multiplier_value
		divide = 10
	}
	add = prestige_level
	add = piety_level
	if = {
		limit = {
			prestige_level >= 3
			has_trait = conqueror
		}
		add = 10
	}
	if = {
		limit = {
			prestige_level >= 4
			has_trait = conqueror
		}
		add = 10
	}
	if = {
		limit = {
			primary_title.tier >= tier_duchy
		}
		add = primary_title.tier
	}
	if = {
		limit = {
			is_ai = no
		}
		add = 1
	}
	if = {
		limit = {
			has_game_rule = random_nickname_frequency_halved
		}
		multiply = 0.5
	}
	else_if = {
		limit = {
			has_game_rule = random_nickname_frequency_quartered
		}
		multiply = 0.25
	}
	else_if = {
		limit = {
			has_game_rule = random_nickname_frequency_doubled
		}
		multiply = 2
	}
}
