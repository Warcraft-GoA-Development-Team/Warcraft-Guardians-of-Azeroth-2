			}

		character = scope:target

		scope:murder_method = flag:riled_up_mob

				}

		}

			trigger = {
				exists = scope:scheme_discovered

					has_trait = humble

				}

	type = character_event
	hidden = yes

	immediate = {
		if = {
			limit = {
				exists = scope:restart_scheme
				exists = scope:scheme
			}
			scope:scheme = {
				add_scheme_progress = -10
			}
		}

			limit = {

			}
			scope:scheme = {
				end_scheme = yes
			}
		}
	}
}
