is_murder_activity_trigger = {
	OR = {
		AND = {
			has_activity_type = activity_wedding
			has_activity_option = {
				category = special_type
				option = wedding_type_bloody
			}
		}
		AND = {
			has_activity_type = activity_feast
			has_activity_option = {
				category = special_type
				option = feast_type_murder
			}
		}
	}
}
