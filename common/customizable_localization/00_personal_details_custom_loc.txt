		trigger = { is_female = yes }
		localization_key = gown
	}

		trigger = { is_female = no }
		localization_key = jacket

		trigger = { intrigue >= decent_skill_rating }

	}
	text = {
		trigger = {
			OR = {

	}
	text = {

		localization_key = died_house_arrest
		trigger = {
			exists = scope:house_arrest_holder
		}
