### Tournament Activity Intents

save_azeroth_intent = {
	icon = save_azeroth_intent

	on_invalidated = {
		trigger_event = activity_system.0090
	}

	ai_will_do = {
		value = 30
		if = {
			limit = { has_trait = brave }
			add = 50
		}
		add = prowess
	}

	scripted_animation = {
		animation = celebrate_sword
	}
}


survive_intent = {
	icon = survive_intent

	#is_valid = { NOT = { has_current_phase = bdp_phase_dark_portal } }

	on_invalidated = {
		trigger_event = activity_system.0090
	}

	ai_will_do = {
		value = 0
		if = {
			limit = {
				is_healthy = no
			}
			add = 200
		}
		if = {
			limit = {
				has_trait = craven
			}
			add = 200
		}
	}

	scripted_animation = {
		animation = fear
	}
}
