light_magic_lifestyle = {
	is_valid = {
		MAGIC_lifestyle_is_valid_trigger = { MAGIC = light }
	}

	is_highlighted = {
		OR = {
			has_trait = magic_good
			has_trait = magic_good_noninheritable
		}
	}

	xp_per_level = 1000
	base_xp_gain = 25
}
