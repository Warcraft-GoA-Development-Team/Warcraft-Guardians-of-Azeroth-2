action_lifestyle_light_magic = {
	type = alert
	check_create_action = {
		
		if = {
			limit = {
				has_focus_light_magic_trigger = yes
				perk_points > 0
				is_adult = yes
			}
			try_create_important_action = {
				important_action_type = action_lifestyle_light_magic
				actor = root
			}
		}
		
	}

	effect = {
		scope:actor = {
			open_view_data = {
				view = lifestyle
			}
		}
	}
}
