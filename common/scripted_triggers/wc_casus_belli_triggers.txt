genocide_cb_can_avenge_culture = {
    scope:defender = {
        any_realm_county = {
            trigger_if = {
                limit = {
                    has_variable = culture_genocided
                }
                var:culture_genocided = scope:attacker.culture
            }
            trigger_else = {
                always = no
            }
        }
    }
}
