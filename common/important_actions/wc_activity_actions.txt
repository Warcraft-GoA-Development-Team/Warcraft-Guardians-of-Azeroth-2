action_can_host_wc_arthas_activity = {
	type = alert
	check_create_action = {
		if = {
			limit = {
				# Make the alert disappear if you are planning an activity already
				NOT = { is_game_view_open = activity_planner }
				can_host_activity = activity_type:activity_arthas_expedition
				should_notify_can_host_activity = activity_type:activity_arthas_expedition
			}
			try_create_important_action = {
				important_action_type = action_can_host_wc_arthas_activity
				activity_type = activity_type:activity_arthas_expedition
			}
		}
		if = {
			limit = {
				# Make the alert disappear if you are planning an activity already
				NOT = { is_game_view_open = activity_planner }
				can_host_activity = activity_type:activity_arthas_expedition_non_dlc
				#should_notify_can_host_activity = activity_type:activity_arthas_expedition_non_dlc
			}
			try_create_important_action = {
				important_action_type = action_can_host_wc_arthas_activity
				activity_type = activity_type:activity_arthas_expedition_non_dlc
			}
		}
	}

	effect = {
		scope:activity_type = { open_view_data = activity_list_detail_host_window }
	}
}
