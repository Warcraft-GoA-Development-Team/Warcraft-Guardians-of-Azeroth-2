# Scarlet mark management.
add_scarlet_mark_tooltip_only_effect = {
	show_as_tooltip = {
		add_trait_force_tooltip = scarlet_mark
		add_character_modifier = {
			modifier = recent_scarlet_mark
			days = 365
		}
	}
}
add_scarlet_mark_actual_effect = {
	add_trait = scarlet_mark
	add_character_modifier = { # Minor negative modifier that stops us from immediately getting our scarlet mark recinded (so you have a guaranteed 'penance period').
		modifier = recent_scarlet_mark
		days = 365
	}
}
