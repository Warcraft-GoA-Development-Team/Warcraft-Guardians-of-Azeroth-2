ep1_09_achievement = {
	possible = {
		exists = global_var:started_true_tolerance_achievement
	}
	happened = {
		custom_description = {
			text = true_tolerance_achievement_trigger
			trigger_if = { # For performance
				limit = {
					any_realm_county = {
						count >= 10
						culture = {
							NOT = { THIS = ROOT.culture }
							cultural_acceptance = {
								target = ROOT.culture
								value >= 95
							}
						}
					}
				}
				any_culture_global = {
					count >= 10
					cultural_acceptance = {
						target = ROOT.culture
						value >= 95
					}
					any_culture_county = { holder.top_liege = ROOT }
				}
			}
			trigger_else = { always = no }
		}
	}
}
