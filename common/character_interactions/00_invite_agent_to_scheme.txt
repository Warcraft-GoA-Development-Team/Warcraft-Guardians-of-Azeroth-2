
invite_agent_to_scheme_interaction = {
	special_interaction = invite_to_scheme_interaction
	hidden = yes
	
	on_accept = {
		scope:recipient = {
			if = {
				limit = {
					OR = {
						always = scope:hook
						always = scope:gift
					}
				}
				force_add_to_scheme = {
					scheme = scope:scheme
					years = 5
				}	
			}
			else = {
				add_to_scheme = scope:scheme
			}
	
		}
		if = {
			limit = { always = scope:gift }
			scope:actor = {
				pay_short_term_gold = {
					gold = bribe_value
					target = scope:recipient
				}
				stress_impact = {
					greedy = minor_stress_impact_gain
				}
			}
		}
		if = {
			limit = { always = scope:hook }
			scope:actor = {
				use_hook = scope:recipient
			}
		}
		scope:actor = {
			stress_impact = {
				paranoid = minor_stress_impact_gain
				honest = minor_stress_impact_gain
			}
		}
	}

	ai_accept = {
		base = 0
		scheme_modifier = {
			scheme = scope:scheme
			scope = scope:recipient			
		}
	}

	auto_accept = {
		custom_description = {
			text = "spending_hook"
			subject = scope:actor
			object = scope:recipient
			scope:hook = yes
			scope:actor = { has_strong_hook = scope:recipient }
		}
	}
	
	send_option = {
		is_valid = {
			scope:actor = {
				gold >= bribe_value
			}
			NOT = {
				scope:recipient = { is_parent_of = scope:scheme.scheme_target }
			}
		}
		flag = gift
		localization = SCHEME_AGENT_GOLD
		current_description = {
			desc = SCHEME_AGENT_GOLD_VALID
		}
	}
	
	#Use hook. Weak hook bonus is added through the scheme's agent join chance, strong hook is in auto-accept
	send_option = {
		is_valid = {
			scope:actor = {
				has_usable_hook = scope:recipient
			}
			NOT = {
				scope:recipient = { is_parent_of = scope:scheme.scheme_target }
			}
		}
		flag = hook
		localization = SCHEME_HOOK
	}
	
	send_options_exclusive = no
	
}
