wc_spell_summon_fire_elemental_rank_1_cost_mana_base_value = 5
wc_spell_summon_fire_elemental_duration_days_base_value = 180
wc_spell_summon_fire_elemental_rank_1_cooldown_days_base_value = wc_spell_cooldown_medium
wc_spell_summon_fire_elemental_rank_2_cooldown_days_base_value = wc_spell_cooldown_long

wc_spell_summon_fire_elemental_cost_mana_value = {
	value = {
		add = {
			value = wc_spell_summon_fire_elemental_rank_1_cost_mana_base_value
			desc = wc_mana_base_value_desc
			format = "WC_MAX_MANA_VALUE_FORMAT"
		}
	}

	if = {
		limit = {
			has_variable = current_spell_rank
		}
		if = {
			limit = {
				var:current_spell_rank = 1
			}
			add = {
				value = {
					add = wc_spell_summon_fire_elemental_rank_1_cost_mana_base_value
				}
				format = "WC_MAX_MANA_VALUE_FORMAT"
				desc = wc_mana_rank_1_value_desc
			}
		}
		else_if = {
			limit = {
				var:current_spell_rank = 2
			}
			add = {
				value = {
					add = wc_spell_summon_fire_elemental_rank_1_cost_mana_base_value
					multiply = wc_spell_rank_2_mana_cost_mult
				}
				format = "WC_MAX_MANA_VALUE_FORMAT"
				desc = wc_mana_rank_2_value_desc
			}
		}
		else_if = {
			limit = {
				var:current_spell_rank = 3
			}
			add = {
				value = {
					add = wc_spell_summon_fire_elemental_rank_1_cost_mana_base_value
					multiply = wc_spell_rank_3_mana_cost_mult
				}
				format = "WC_MAX_MANA_VALUE_FORMAT"
				desc = wc_mana_rank_3_value_desc
			}
		}
		
	}

	if = {
		limit = {
			has_variable = current_spell_elemental_type
		}
		switch = {
			trigger = var:current_spell_elemental_type

			flag:spirit = {
				add = {
					value = wc_spell_summon_fire_elemental_rank_1_cost_mana_base_value
					multiply = 0.25
				}
				format = "WC_MAX_MANA_VALUE_FORMAT"
				desc = wc_mana_decay_value_desc
			}

			flag:decay = {
				
				add = {
					value = wc_spell_summon_fire_elemental_rank_1_cost_mana_base_value
					multiply = -0.5
				}
				format = "WC_MAX_MANA_VALUE_FORMAT"
				desc = wc_mana_decay_value_desc
			}
		}
	}
}

wc_spell_summon_fire_elemental_cooldown_value = {
	if = {
		limit = {
			has_variable = current_spell_rank
		}
		if = {
			limit = {
				var:current_spell_rank = 1
			}
			add = {
				value = {
					add = wc_spell_summon_fire_elemental_rank_1_cooldown_days_base_value
				}
				format = "WC_MAX_MANA_VALUE_FORMAT"
				desc = wc_mana_rank_1_value_desc
			}
		}
		else_if = {
			limit = {
				var:current_spell_rank = 2
			}
			add = {
				value = wc_spell_summon_fire_elemental_rank_2_cooldown_days_base_value
				format = "WC_MAX_MANA_VALUE_FORMAT"
				desc = wc_mana_rank_2_value_desc
			}
		}
		else_if = {
			limit = {
				var:current_spell_rank = 3
			}
			add = {
				value = wc_spell_summon_fire_elemental_rank_2_cooldown_days_base_value
				format = "WC_MAX_MANA_VALUE_FORMAT"
				desc = wc_mana_rank_3_value_desc
			}
		}
		
	}
	if = { # spell_cd_mult should be positive. if it's 0.8 and the cd time was 100, new cd time is 80
		limit = { has_variable = spell_cd_mult }
		multiply = {
			value = var:spell_cd_mult 
			format = "WC_MAX_MANA_VALUE_FORMAT"
			desc = DEFAULT_TEXT
		}
	}
}
