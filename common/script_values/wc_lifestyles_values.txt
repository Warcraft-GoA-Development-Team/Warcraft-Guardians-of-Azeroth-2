wc_magic_lifestyle_focus_selection_weight_multiply_all_magic_value = { # Common multiplier of all focuses
	value = 1

	if = {
		limit = {
			OR = {
				has_trait = magic_bad
				has_trait = magic_bad_noninheritable
			}
		}
		multiply = 0.01
	}
	if = {
		limit = { # Don't upgrade historical characters' classes on gamestart
			game_start_day_trigger = yes
		}
		multiply = 0
	}
}
