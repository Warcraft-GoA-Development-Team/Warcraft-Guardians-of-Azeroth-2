can_be_wounded_trigger = {
	NOT = {
		has_trait_rank = {
			trait = wounded
			rank = 3
		}
	}
}
can_be_maimed_trigger = {
	NAND = {
		has_trait = one_legged
		has_trait = disfigured
		has_trait = one_eyed
		has_trait = maimed
	}
}
knight_can_be_resurrected_trigger = {
	can_become_undead_trigger = yes

	exists = var:battle_death_enemy
	OR = {
		var:battle_death_enemy = { government_has_flag = government_resurrect_knights }
		scope:enemy_knight = { can_use_frostmourne_to_twist_soul_trigger = yes }
	}
}
